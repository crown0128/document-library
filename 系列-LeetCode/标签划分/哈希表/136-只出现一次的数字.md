## <a name="chapter-three" id="chapter-three"></a>三 解题思路

> [返回目录](#chapter-one)

Set：

```js
/**
 * @param {number[]} nums
 * @return {number}
 */
const singleNumber = (nums) => {
  // 1. 设置集合
  const set = new Set();

  // 2. 遍历数组
  for (let i = 0; i < nums.length; i++) {
    // 如果这个值重复出现了，咱们清空它
    if (set.has(nums[i])) {
      set.delete(nums[i]);
    } else { // 否则添加进来
      set.add(nums[i]);
    }
  }

  // 3. 转数组，取唯一值
  return [...set].pop();
};

console.log(singleNumber([4, 1, 2, 1, 2])); // 4
```
