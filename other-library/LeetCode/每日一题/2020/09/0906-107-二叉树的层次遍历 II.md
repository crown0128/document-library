## <a name="chapter-three" id="chapter-three"></a>三 解题思路

> [返回目录](#chapter-one)

广度优先搜索：

```js
/**
 * Definition for a binary tree node.
 * function TreeNode(val) {
 *     this.val = val;
 *     this.left = this.right = null;
 * }
 */
/**
 * @param {TreeNode} root
 * @return {number[][]}
 */
const levelOrderBottom = (root) => {
  // 狗不理
  if (!root) {
    return [];
  }

  // 1. 设置结果集
  const result = [];

  // 2. 设置当前层
  let nowRoot = [root];

  // 3. 广度优先搜索
  while (nowRoot.length) {
    // 3.1 设置下一层
    const nextRoot = [];

    // 3.2 设置当前层元素的获取
    const nowFloor = [];

    // 3.3 遍历当前层，获取下一层的元素以及当前层的值
    for (let i = 0; i < nowRoot.length; i++) {
      
      // 3.3.1 获取当前层的值
      nowFloor.push(nowRoot[i].val);

      // 3.3.2 设置下一层的值
      if (nowRoot[i].left) {
        nextRoot.push(nowRoot[i].left);
      }
      if (nowRoot[i].right) {
        nextRoot.push(nowRoot[i].right);
      }
    }

    // 3.4 进行下一层和当前层的交接，以及赋值
    nowRoot = nextRoot;
    result.unshift(nowFloor);
  }

  // 4. 返回最终结果
  return result;
};

const root = {
  val: 3,
  left: { val: 9, left: null, right: null },
  right: {
    val: 20,
    left: { val: 15, left: null, right: null },
    right: { val: 7, left: null, right: null },
  },
};

console.log(levelOrderBottom(root));
```
