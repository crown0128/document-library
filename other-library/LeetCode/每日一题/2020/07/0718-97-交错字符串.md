## <a name="chapter-three" id="chapter-three"></a>三 解题思路

> [返回目录](#chapter-one)

动态规划：

```js
/**
 * @param {string} s1
 * @param {string} s2
 * @param {string} s3
 * @return {boolean}
 */
const isInterleave = (s1, s2, s3) => {
  const n = s1.length,
    m = s2.length,
    t = s3.length;
  if (n + m !== t) {
    return false;
  }

  const f = Array.from(Array(n + 1), () => Array.from(Array(m + 1), () => false));

  f[0][0] = true;
  for (let i = 0; i <= n; i++) {
    for (let j = 0; j <= m; j++) {
      const p = i + j - 1;
      if (i > 0) {
        f[i][j] = f[i][j] || (f[i - 1][j] && s1[i - 1] === s3[p]);
      }
      if (j > 0) {
        f[i][j] = f[i][j] || (f[i][j - 1] && s2[j - 1] === s3[p]);
      }
    }
  }
  
  return f[n][m];
};

console.log(isInterleave(
  'aabcc',
  'dbbca',
  'aadbbcbcac',
)); // true
```

> 动态规划之滚动数组

```js
/**
 * @param {string} s1
 * @param {string} s2
 * @param {string} s3
 * @return {boolean}
 */
const isInterleave = (s1, s2, s3) => {
  const n = s1.length,
    m = s2.length,
    t = s3.length;
  if (n + m !== t) {
    return false;
  }

  const f = Array.from(Array(n + 1), () => false);

  f[0] = true;
  for (let i = 0; i <= n; i++) {
    for (let j = 0; j <= m; j++) {
      const p = i + j - 1;
      if (i > 0) {
        f[j] &= (s1[i - 1] === s3[p]);
      }
      if (j > 0) {
        f[j] |= (f[j - 1] && s2[j - 1] === s3[p]);
      }
    }
  }

  return f[m];
};

console.log(isInterleave(
  'aabcc',
  'dbbca',
  'aadbbcbcac',
)); // true
```
