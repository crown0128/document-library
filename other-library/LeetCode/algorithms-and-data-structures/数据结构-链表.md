  // 向列表尾部添加一个新的项。
  this.append = function(element) {
    let node = new Node(element);
    let current;
    if (!head) {
      head = node;
      console.log(head);
    } else {
      current = head;
      // 循环列表，直到找到最后一项
      while (current.next) {
        current = current.next;

      current.next = node;
